is_alone_or_in_faction_at_peace_with_RME = {
	OR = {
		is_in_faction = no
		is_in_faction_with = RME
		AND = {
			is_in_faction = yes
			NOT = {
				is_in_faction_with = RME
			}
		}
	}
	NOT = {
		any_other_country = {
			is_in_faction_with = PREV
			has_war_with = RME
		}
	}
}

is_owned_by_ROOT_or_subject = {
	custom_trigger_tooltip = {
		tooltip = is_owned_by_ROOT_or_subject
		OWNER = {
			OR = {
				tag = ROOT
				is_subject_of = ROOT
			}
		}
	}
}

	custom_trigger_tooltip = {

			AND = {
				is_subject = no
				is_in_faction_with = ROOT
			}
			is_subject_of = ROOT
		}
	}
}
