add_namespace = tesb_GER

#Bohemian occupation
country_event = {
	id = tesb_GER.10
	title = tesb_GER.10.t
	desc = tesb_GER.10.d
	picture = GFX_report_event_german_italian_pact
	fire_only_once = yes

	trigger = {
		tag = GER
		has_country_flag = GER_election_1936_flag
		NOT = { has_country_flag = GER_election_event }
	}

	mean_time_to_happen = {
		days = 15
	}

	immediate = {
		set_country_flag = {
			flag = GER_election_event
			days = 15
		}
	}
	
	#The Bohemians must be free!
	option = {
		name = tesb_GER.10.a
		add_popularity = {
			ideology = fascism
			popularity = 0.03
		}
		add_popularity = {
			ideology = communism
			popularity = 0.025
		}
		add_war_support = 0.05
	}

	#Let's not stir up tension
	option = {
		name = tesb_GER.10.b
		add_popularity = {
			ideology = democratic
			popularity = 0.025
		}
		add_stability = -0.05
	}
}
